18   */
19  public class TestServerAuthModuleAuthConfig implements

21  
22      /**
23       * <p>
24       * The {@link javax.security.auth.message.MessageInfo} map must contain this
25       * key and its associated value, if and only if authentication is required
26       * to perform the resource access corresponding to the HttpServletRequest to
27       * which the ServerAuthContext will be applied. Authentication is required
28       * if use of the HTTP method of the HttpServletRequest at the resource
29       * identified by the HttpServletRequest is covered by a Servlet auth
30       * constraint, or in a JSR 115 compatible runtime, if the corresponding
31       * WebResourcePermission is NOT granted to an unauthenticated caller. In a
32       * JSR 115 compatible runtime, the corresponding WebResourcePermission may
33       * be constructed directly from the HttpServletRequest as follows:
34       * </p>
35       * <blockquote> public WebResourcePermission(HttpServletRequest
36       * request); </blockquote> *
37       * <p>
38       * The authentication context configuration system must use the value of
39       * this property to establish the corresponding value within the
40       * requestPolicy passed to the authentication modules of the
41       * {@link javax.security.auth.message.config.ServerAuthContext} acquired to
42       * process the {@link javax.security.auth.message.MessageInfo}.
43       * </p>
44       */
45      private static final String JAVAX_SECURITY_AUTH_MESSAGE_MESSAGE_POLICY_IS_MANDATORY = "javax.security.auth.message.MessagePolicy.isMandatory";
46  
47      /**
48       * Mandatory message policy.
49       */
50      protected static final MessagePolicy MANDATORY = new MessagePolicy(new TargetPolicy[0], true);
51  
52      /**
53       * Non-mandatory message policy.
54       */
55      protected static final MessagePolicy NON_MANDATORY = new MessagePolicy(new TargetPolicy[0], false);
56  
57      /**
58       * Application context.
59       */
60      private final String appContext;
61  
62      /**
63       * Callback handler.
64       */
65      private final CallbackHandler handler;
66  
67      /**
68       * Layer. Usually HttpServlet or SOAPMessage.
69       */
70      private final String layer;
71  
72      /**
73       * Setup options.
74       */
75      private final Map<String, String> options;
76  
77      /**
78       * Builds the config.
79       *
80       * @param options
81       *            options
82       * @param layer
83       *            layer
84       * @param appContext
85       *            application context
86       * @param handler
87       *            callback handler
88       */
89      public TestServerAuthModuleAuthConfig(final Map<String, String> options,
90          final String layer,
91          final String appContext,
92          final CallbackHandler handler) {
93  
94          this.appContext = appContext;
95          this.layer = layer;
96          this.options = options;
97          this.handler = handler;
98      }
99  
100     /**
101      * Augments the properties with additional properties.
102      *
103      * @param properties
104      *            properties to augment with.
105      * @return augmented properties
106      */
107     @SuppressWarnings("unchecked")
108     protected Map<?, ?> augmentProperties(@SuppressWarnings("rawtypes") final Map properties) {
109 
110         if (properties == null) {
111             return options;
112         }
113         final ConcurrentMap<?, ?> augmentedOptions = new ConcurrentHashMap<>(options);
114         augmentedOptions.putAll(properties);
115         return augmentedOptions;
116 
117     }
118 
119     /**
120      * {@inheritDoc}
121      */
122     @Override
123     public String getAppContext() {
124 
125         return appContext;
126     }
127 
128     /**
129      * {@inheritDoc}
130      * <p>
131      * Augments the options with the properties specified and initializes the
132      * module as mandatory or non-mandatory depending on whether the
133      * authContextID is <code>null</code>.
134      * </p>
135      */
136     @Override
137     public ServerAuthContext getAuthContext(final String authContextID,
138         final Subject serviceSubject,
139         @SuppressWarnings("rawtypes") final Map properties) throws AuthException {
140 
141         final Map<?, ?> augmentedOptions = augmentProperties(properties);
142         final TestServerAuthModule context = new TestServerAuthModule();
143 
144         if (authContextID == null) {
145             context.initialize(NON_MANDATORY, NON_MANDATORY, handler, augmentedOptions);
146         } else {
147             context.initialize(MANDATORY, MANDATORY, handler, augmentedOptions);
148         }
149         return context;
150     }
151 
152     /**
153      * Checks for the presence of
154      * {@value #JAVAX_SECURITY_AUTH_MESSAGE_MESSAGE_POLICY_IS_MANDATORY} in the
155      * map.
156      *
157      * @param messageInfo
158      *            contains the message request, response and some system
159      *            populated map.
160      * @return the string representation of the {@link MessageInfo} if it is
161      *         mandatory, <code>null</code> otherwise.
162      */
163     @Override
164     public String getAuthContextID(final MessageInfo messageInfo) {
165 
166         final Object isMandatory = messageInfo.getMap()
167             .get(JAVAX_SECURITY_AUTH_MESSAGE_MESSAGE_POLICY_IS_MANDATORY);
168         if (isMandatory != null && isMandatory instanceof String && Boolean.valueOf((String) isMandatory)) {
169             return messageInfo.toString();
170         }
171         return null;
172     }
