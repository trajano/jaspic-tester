1   package net.trajano.auth;
2   
3   import java.util.HashMap;
4   import java.util.Map;
5   
6   import javax.security.auth.message.config.AuthConfigFactory;
7   import javax.servlet.ServletContextEvent;
8   import javax.servlet.ServletContextListener;
9   
10  public class Initializer implements
11      ServletContextListener {
12  
13      private String registration;
14  
15      /**
16       * <p>
17       * Removes registration from the factory.
18       * </p>
19       * {@inheritDoc}
20       */
21      @Override
22      public void contextDestroyed(final ServletContextEvent sce) {
23  
24          AuthConfigFactory.getFactory().removeRegistration(registration);
25      }
26  
27      /**
28       * <p>
29       * Registers the configuration into the factory.
30       * </p>
31       * {@inheritDoc}
32       */
33      @Override
34      public void contextInitialized(final ServletContextEvent sce) {
35  
36          final Map<String, String> options = new HashMap<>();
37          registration = AuthConfigFactory.getFactory().registerConfigProvider(AuthModuleConfigProvider.class.getName(),
38              options, "HttpServlet", null, null);
39  
40      }
41  
42  }
